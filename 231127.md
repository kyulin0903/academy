# 보통 카운팅 하는 법
```
// Online C compiler to run C program online
#include <stdio.h>

int main() {
    
    int test[8] = {2, 4, 5, 6, 2, 5, 5, 7};
    int count[9] = {0, };
    for(int i = 0; i < 8; i++)
    {
        count[test[i]]++;
    }
    for(int i = 0; i < 8; i++)
    {
        printf("%d : %d\n", i, count[i]);
    }
    
    return 0;
}
```
